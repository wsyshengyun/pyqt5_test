function parseHelpData(data){
    if (helpHttp != null) {
        if (4 == helpHttp.readyState) {
            if (200 == helpHttp.status) {
                var rows = helpHttp.responseText.split("\r\n");
                var head = rows[0].split("\t");
                var enIndex = null;

                for(var i=0; i<head.length; i++){
                    if(head[i] == "en"){}
                    enIndex = i;
                }

                for(var i=1; i<rows.length; i++){
                    if(rows[i] != null && rows[i].length != 0){
                        var item = rows[i].split("\t");
                        var helps = {};

                        for(var j=1; j<head.length; j++){
                            var help = {};
                            if(item[j] != null && item[j].length != 0){
                                var contents = item[j].split("\n");

                            }else{
                                var contents = item[enIndex].split("\n");
                            }
                            for(var n=0; n<contents.length; n++){
                                var content = contents[n];
                                if(content != null && content.length != 0 ){
                                    if(-1 != content .search("Łş"))
                                    {
                                        var conArr = content.split("Łş");
                                    }
                                    else
                                    {
                                        var conArr = content.split(":");
                                    }

                                    var conVal="";
                                    for(var m=1; m<conArr.length; m++){
                                        if(m == 1){
                                            conVal = conArr[m];
										}else{
                                            conVal = conVal + ":" + conArr[m];
										}

                                    }
                                    help[conArr[0]] = conVal;
                                }
                            }
                            helps[head[j]] = help;
                        }
                        helpsObj[item[0]] = helps;
                    }
                }

                return helpsObj;
            }
        }
    }
}

function xmlHookFun() {
	if (xmlHttp != null) {
		if (4 == xmlHttp.readyState) {
			if (200 == xmlHttp.status) {
				var rows = xmlHttp.responseText.split("\r\n");			
				var colsLen = rows[0].split("\t").length;
				gLangId.addItem("MAX_COLS", colsLen - 1);
				for (var i=0; i<rows.length; i++) {
					if (rows[i]) {
						var cols = rows[i].split("\t");
						if (i != 0) gStrList[i - 1] = new Array();
						for (var j=0; j<colsLen; j++) {
							if (i == 0 && j != 0) {
								gLangId.addItem(cols[j], j - 1);
							} else {
								if (j == 0) gStrId.addItem(cols[j], i - 1);	
								else 		gStrList[i - 1][j - 1] = cols[j];
							}
						}
					}
				}
			}
		}
	}
}
